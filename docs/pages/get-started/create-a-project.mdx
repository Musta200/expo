    <h1>Copy Text App</h1>
    <textarea id="textToCopy" placeholder="Type something here..."></textarea><br>
    <button onclick="copyText()">Copy Text</button>

    <p id="message" style="color: green; margin-top: 20px;"></p>

    <script>
        function copyText() {
            // Get the text area element
            const textArea = document.getElementById('textToCopy');
            
            // Select the text
            textArea.select();
            textArea.setSelectionRange(0, 99999); // For mobile devices
            
            // Copy the text to clipboard
            navigator.clipboard.writeText(textArea.value)
                .then(() => {
                    document.getElementById('message').innerText = "Text copied to clipboard!";
                })
                .catch(err => {
                    document.getElementById('message').innerText = "Failed to copy text.";
                    console.error("Error copying text: ", err);
                });
        }
    </script>
